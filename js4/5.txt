<script>
  const list = document.getElementById("books");
  let active = null;

  list.addEventListener("click", (e) => {
    if (e.target.tagName !== "LI") return;

    if (active) active.classList.remove("active");

    active = e.target;
    active.classList.add("active");
  });
</script>
